4.Run KMeans for multiple values of K and store inertia values.

In[5]:
from sklearn.cluster import KMeans
import pandas as pd

# Define the range of K values to test
k_range = range(1, 11)
inertia_values = []

# Fit KMeans for each k in the range
for k in k_range:
    # random_state=42 ensures reproducibility
    kmeans = KMeans(n_clusters=k, random_state=42, n_init=10)
    kmeans.fit(df_scaled)
    inertia_values.append(kmeans.inertia_)

# Store the results in a DataFrame for easy viewing
inertia_df = pd.DataFrame({'K': list(k_range), 'Inertia': inertia_values})
print(inertia_df)

import pandas as pd
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans

# 1. Load the dataset
df = pd.read_csv("C:/Users/waghu/OneDrive/Desktop/ELEVATE Labs/customer_segmentation.csv")

# 2. Data Cleaning: Drop rows with missing values in 'Income'
df_cleaned = df.dropna(subset=['Income']).copy()

# 3. Feature Selection: Select numerical columns for clustering
# Excluding IDs and constant columns (Z_CostContact, Z_Revenue)
cols_to_use = [
    'Year_Birth', 'Income', 'Kidhome', 'Teenhome', 'Recency', 
    'MntWines', 'MntFruits', 'MntMeatProducts', 'MntFishProducts', 
    'MntSweetProducts', 'MntGoldProds', 'NumDealsPurchases', 
    'NumWebPurchases', 'NumCatalogPurchases', 'NumStorePurchases', 
    'NumWebVisitsMonth'
]
X = df_cleaned[cols_to_use]

# 4. Scaling: Normalize the features
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# 5. Clustering: Apply K-Means
# We use 4 clusters as an example
kmeans = KMeans(n_clusters=4, random_state=42, n_init=10)
cluster_labels = kmeans.fit_predict(X_scaled)

# 6. Add labels back: Create a new column 'Cluster' in the dataframe
df_cleaned['Cluster'] = cluster_labels

# 7. Save the updated dataset
df_cleaned.to_csv('customer_segmentation_with_clusters.csv', index=False)

print("Cluster labels added successfully. Sample output:")
print(df_cleaned[['ID', 'Income', 'Cluster']].head())
